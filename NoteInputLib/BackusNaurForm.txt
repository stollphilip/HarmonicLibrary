<score> ::= <line>+
<line> ::= <command> [<command-separator> <command>]*
<command> ::= <command-name> [[<symbol-separator>] <command-argument>]*
<command> ::= <measure> | <time> | <key> | <part> | <chord>
<part> ::= <part-name> (<note>|<rest>)+
<chord> ::= (<pitch>|<rest>)+ [<duration>]
<note> ::= <pitch> [<duration>] [<tie>]
<pitch> ::= <height> | <pitch-letter> <octave>
<height> ::= number [<accidental>] | '('number [<accidental>] ')'
<accidental> ::= '#' | 'b' | 'n' | "sharp" | "flat" | "natural"
<duration> ::= <beats>[:<numerator>:<denominator>]
<rest> ::= 'r' [<duration>]
<tie> ::= '^'
<meter> ::= <beats>[:<numerator>[:<denominator>]]
<time> ::= <meter>
<measure> ::= "measure" [number] [<meter>] [(<part>)+|(<chord>)]
<key> ::=  'A'|'B'|'C'|'D'|'E'|'F'|'G' [<accidental>] "Major"
<comment> ::= "//" <text>
<command> ::= <token> [<space>] <number>+ [<text>]
<height> ::= signed integer
<pitch-letter> ::= 'A'|'B'|'C'|'D'|'E'|'F'|'G'
<octave> ::= natural number
<command-separator> ::= ','
<symbol-separator> ::= ' '
